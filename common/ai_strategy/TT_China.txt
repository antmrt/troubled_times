ZHI_prepare_war = {
	allowed = {
		tag = ZHI
	}
	abort_when_not_enabled = yes
	enable = {
		OR = {
			ZHI = {
				has_war_with = ANH
			}
			ANH = {
				check_variable = {
					num_owned_controlled_states > 3
				}
			}

		}
	}
	ai_strategy = {
		type = front_unit_request
		TAG = ANH
		value = 3000 
	}
    ai_strategy = {
		type = prepare_for_war
		id = ANH
		value = 1000
	}
	ai_strategy = {
		type = consider_weak
		id = ANH
		value = 1000
	}
	ai_strategy = {
		type = declare_war
		id = ANH
		value = 1000
	}
	ai_strategy = {
        type = force_build_armies
        value = 100
    }
}
